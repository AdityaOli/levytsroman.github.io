{% highlight ruby %} def quick_sort(arr, l, r) return arr if l > r idx = sort(arr, l, r) quick_sort(arr, l, idx - 1) quick_sort(arr, idx + 1, r) end def sort(arr, l, r) i = l + 1 j = l + 1 p = arr[l] while i <= r && j <= r if arr[i] < p arr[i],arr[j] = arr[j],arr[i] j += 1 end i += 1 end arr[j - 1],arr[l] = arr[l],arr[j - 1] return j - 1 end {% endhighlight %}
